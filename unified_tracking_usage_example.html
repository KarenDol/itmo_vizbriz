Unified Tracking Usage Example

This page demonstrates how to use the unified tracking utility in your frontend applications.

1. Include the Tracking Script

<script src="/static/js/unified_tracking.js"></script>

2. Initialize and Set Quiz Context

// Generate quiz ID and set context
async function initializeTracking() {
    const quizId = await unifiedTracker.generateQuizId();
    unifiedTracker.setQuizContext(
        quizId, 
        'patient@example.com', 
        'basic_quiz', 
        1
    );
}

3. Track Page Views

// Track page view
await unifiedTracker.trackPageView('stage_a_step_1', {
    step_number: 1,
    time_spent: 45
});

4. Track CTA Clicks

// Track CTA click
await unifiedTracker.trackCTAClick(
    'schedule_sleep_test',
    'Schedule My Sleep Test Now',
    'results_page',
    false
);

5. Track Form Submissions

// Track form submission
await unifiedTracker.trackFormSubmit('consultation_request', {
    form_fields: ['name', 'email', 'phone'],
    completion_time: 120
});

6. Get Current Context

// Get current tracking context
const context = unifiedTracker.getContext();
console.log(context);

Integration with Quiz Flow

Typical Quiz Integration:
1. Generate quiz ID when quiz starts
2. Track each page view as user progresses
3. Track CTA clicks for conversion events
4. Track form submission when quiz completes 
// Complete quiz flow example
async function startQuiz() {
    // 1. Generate quiz ID
    const quizId = await unifiedTracker.generateQuizId();
    unifiedTracker.setQuizContext(quizId, patientEmail, 'basic_quiz', clinicId);
    
    // 2. Track quiz start
    await unifiedTracker.trackPageView('quiz_start');
    
    // 3. Show first question
    showQuestion(1);
}

async function nextQuestion(questionNumber) {
    // Track page view for each question
    await unifiedTracker.trackPageView(`question_${questionNumber}`);
    showQuestion(questionNumber + 1);
}

async function handleCTAClick(ctaType, ctaText) {
    // Track CTA clicks
    await unifiedTracker.trackCTAClick(ctaType, ctaText, 'results_page');
    // Handle the CTA action
}

async function submitQuiz() {
    // Track quiz completion
    await unifiedTracker.trackFormSubmit('quiz_submission', {
        total_questions: 10,
        completion_time: 300
    });
    // Submit quiz data
}
